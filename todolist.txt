1.resize root:
	(1)df -h
	(2)lsblk
	(3)parted /dev/mmcblk2
	(4)mkpart primary ext4 2GB 100%
	(5)mkfs.ext4 /dev/mmcblk0p2
	(6)mkdir /mnt/newspace
	(7)mount /dev/mmcblk0p2 /mnt/newspace
	(8)cp -a /overlay/* /mnt/newspace/
	(9)umount /mnt/newspace
	(10)# 修改 /etc/config/fstab 添加：
		config mount
  		option target '/overlay'
  		option device '/dev/mmcblk0p2'
  		option fstype 'ext4'
  		option options 'rw,sync'
  		option enabled '1'
